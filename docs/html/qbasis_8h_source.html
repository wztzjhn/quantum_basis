1027
1028 // m step of Lanczos (for iram: m = k + np; for simple Lanczos, m depends on the convergence speed)
1029 // on entry, assuming k steps of Lanczos already performed
1030 // v of length m+1, hessenberg matrix of size m*m (m-step Lanczos)
1031 // after decomposition, mat * v[0:m-1] = v[0:m-1] * hessenberg + b[m] * v[m] * e_m^T,
1032 // where e_m has only one nonzero element: e[0:m-2] == 0, e[m-1] = 1
1033 //
1034 // maxit: maximum allowed Lanczos steps, m < maxit
1035 // maxit also serves as the leading dimension of hessenberg matrix
1036 // a[j] = hessenberg[j+maxit], diagonal of hessenberg matrix
1037 // b[j] = hessenberg[j]
1038 // a[0] b[1] -> note: b[0] not used
1039 // b[1] a[1] b[2]
1040 // b[2] a[2] b[3]
1041 // b[3] a[3] b[4]
1042 // .. .. .. b[k-1]
1043 // b[k-1] a[k-1]
1044 // on entry:
1045 // a[0], ..., a[k-1] in hessenberg matrix
1046 // b[0], ..., b[k-1], b[k] in hessenberg matrix
1047 // on exit:
1048 // a[0], ..., a[m-1] in hessenberg matrix
1049 // b[0], ..., b[m-1], b[m] in hessenberg matrix
1050 //
1051 // if purpose == "iram":
1052 // on entry:
1053 // v[0], ..., v[k] stored in v
1054 // on exit:
1055 // v[0], ..., v[m] stored in v
1056 //
1057 // if purpose == "sr_val0" (smallest eigenvalue), or purpose == "dnmcs":
1058 // on entry:
1059 // v[k-1], v[k] stored in v. If k%2==0, stored as {v[k],v[k-1]}; else stored as {v[k-1],v[k]}
1060 // on exit:
1061 // v[m-1], v[m] stored in v. If m%2==0, stored as {v[m],v[m-1]}; else stored as {v[m-1],v[m]}
1062 //
1063 // if purpose == "sr_vec0" (smallest eigenvector):
1064 // same as "sr_val0", but with one extra column storing the eigenvector
1065 template <typename T, typename MAT>
1066 void lanczos(MKL_INT k, MKL_INT np, const MKL_INT &maxit, MKL_INT &m, const MKL_INT &dim,
1067 const MAT &mat, T v[], double hessenberg[], const std::string &purpose);

1070 // given well converged eigenvalue E0, and a good initital guess v[0], solves (H - E0) * v[j] = 0
1071 // on entry: assuming m-step finished, v contains the initital guess v[m]
1072 // r[0] = -(H-E0)*v[0], p[0] = r[0]
1073 // on exit: v rewritten by the converged solution
